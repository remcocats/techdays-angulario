
Step 4: Consume the service in the component

import {Component, OnInit} from '@angular/core';
import {EventService} from './event.service';
import {ChangeDetectorRef} from '@angular/core';

export class AppComponent implements OnInit {
  public title = 'app';
  public eventList = new Array();

  constructor(private ref: ChangeDetectorRef, private eventService: EventService) {}

  ngOnInit() {
    this.eventService.getEvents('http://localhost/api').subscribe(event => {
      console.log(event);
      this.eventList.push(event);
      this.ref.detectChanges();
    });
  }
}

Inject the service into the component and inject the reference to the angular change detector.
Make the class implement OnInit interface.
In the initialization method subscribe to the Observable and update the state and notify the changeDetector.
